# My Chefs Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Let users build and persist a personal list of up to 6 YouTube cooking creators, then search those channels exactly like the existing Chef's Table feature.

**Architecture:** A new `customChefsList` Convex table stores one document per anonymous session (keyed by `sessionId`) containing an array of up to 6 custom chef entries. A new `/my-chefs` page handles adding/removing chefs via URL input. The home page gains a 3rd "My Chefs" tab that reads the saved list and passes it to the existing `searchChefVideos` action — no changes to the search/results path.

**Tech Stack:** Next.js App Router, TypeScript, Tailwind CSS, Convex (mutations/queries/actions), YouTube Data API v3, Vitest (unit + integration via `convex-test`), Playwright (E2E)

```typescript
// ─── addCustomChef ─────────────────────────────────────────────────────────

const MAX_CHEFS = 6;

export const addCustomChef = mutation({
  args: {
    sessionId:        v.string(),
    channelId:        v.string(),
    channelName:      v.string(),
    channelThumbnail: v.string(),
    resolvedAt:       v.number(),
  },
  handler: async (ctx, args) => {
    const doc = await ctx.db
      .query("customChefsList")
      .withIndex("by_session", (q) => q.eq("sessionId", args.sessionId))
      .unique();

    const newChef = {
      channelId:        args.channelId,
      channelName:      args.channelName,
      channelThumbnail: args.channelThumbnail,
      addedAt:          Date.now(),
      resolvedAt:       args.resolvedAt,
    };

    if (!doc) {
      await ctx.db.insert("customChefsList", {
        sessionId: args.sessionId,
        chefs:     [newChef],
        updatedAt: Date.now(),
      });
      return;
    }

    if (doc.chefs.length >= MAX_CHEFS) {
      throw new Error("limit_reached");
    }

    if (doc.chefs.some((c) => c.channelId === args.channelId)) {
      throw new Error("duplicate");
    }

    await ctx.db.patch(doc._id, {
      chefs:     [...doc.chefs, newChef],
      updatedAt: Date.now(),
    });
  },
});
```

```typescript
// lib/parseYouTubeUrl.ts
// Pure function — no browser or Node.js APIs. Safe to import from Convex actions.

export type ParseResult =
  | { type: "handle"; value: string }
  | { type: "channelId"; value: string }
  | { type: "error" };

export function parseYouTubeInput(input: string): ParseResult {
  const trimmed = input.trim();
  if (!trimmed) return { type: "error" };

  // Bare @handle: e.g. "@babish"
  if (/^@[\w.-]+$/.test(trimmed)) {
    return { type: "handle", value: trimmed.slice(1) };
  }

  try {
    const urlStr = trimmed.startsWith("http") ? trimmed : `https://${trimmed}`;
    const url = new URL(urlStr);

    if (!url.hostname.endsWith("youtube.com")) {
      return { type: "error" };
    }

    // youtube.com/@handle
    const handleMatch = url.pathname.match(/^\/@([\w.-]+)/);
    if (handleMatch) {
      return { type: "handle", value: handleMatch[1] };
    }

    // youtube.com/channel/UCxxxxxx
    const channelMatch = url.pathname.match(/^\/channel\/(UC[\w-]+)/);
    if (channelMatch) {
      return { type: "channelId", value: channelMatch[1] };
    }

    return { type: "error" };
  } catch {
    return { type: "error" };
  }
}
```

```typescript
// ─── resolveYouTubeChannel ─────────────────────────────────────────────────

export const resolveYouTubeChannel = action({
  args: { input: v.string() },
  handler: async (_ctx, args) => {
    const apiKey = process.env.YOUTUBE_API_KEY;
    if (!apiKey) {
      return { ok: false as const, error: "api_error" as const };
    }

    const parsed = parseYouTubeInput(args.input);
    if (parsed.type === "error") {
      return { ok: false as const, error: "parse_error" as const };
    }

    try {
      const url = new URL("https://www.googleapis.com/youtube/v3/channels");
      url.searchParams.set("key", apiKey);
      url.searchParams.set("part", "snippet");

      if (parsed.type === "handle") {
        url.searchParams.set("forHandle", parsed.value);
      } else {
        url.searchParams.set("id", parsed.value);
      }

      const response = await fetch(url.toString());
      const data = await response.json();

      if (data.error || !data.items?.length) {
        return { ok: false as const, error: "not_found" as const };
      }

      const channel = data.items[0];
      return {
        ok:               true as const,
        channelId:        channel.id as string,
        channelName:      channel.snippet.title as string,
        channelThumbnail: channel.snippet.thumbnails.default.url as string,
        resolvedAt:       Date.now(),
      };
    } catch {
      return { ok: false as const, error: "api_error" as const };
    }
  },
});
```

**Step 1: Create `app/my-chefs/page.tsx`**

```tsx
"use client";
import { useState } from "react";
import { useQuery, useMutation, useAction } from "convex/react";
import Link from "next/link";
import { api } from "@/convex/_generated/api";
import { getSessionId } from "@/lib/session";
import { CustomChefCard } from "@/components/CustomChefCard";

const MAX_CHEFS = 6;

type ResolveError = "parse_error" | "not_found" | "api_error" | null;

type PreviewChef = {
  channelId:        string;
  channelName:      string;
  channelThumbnail: string;
  resolvedAt:       number;
};

function errorMessage(error: ResolveError): string {
  if (error === "parse_error") return "Paste a YouTube channel URL or @handle";
  if (error === "not_found")   return "We couldn't find that channel — check the URL";
  return "Something went wrong — try again";
}

export default function MyChefsDPage() {
  const sessionId = getSessionId();

  const chefs       = useQuery(api.customChefs.listCustomChefs, { sessionId }) ?? [];
  const addChef     = useMutation(api.customChefs.addCustomChef);
  const removeChef  = useMutation(api.customChefs.removeCustomChef);
  const resolveChannel = useAction(api.customChefs.resolveYouTubeChannel);

  const [input,      setInput]      = useState("");
  const [preview,    setPreview]    = useState<PreviewChef | null>(null);
  const [resolveErr, setResolveErr] = useState<ResolveError>(null);
  const [isResolving, setIsResolving] = useState(false);
  const [addError,   setAddError]   = useState<string | null>(null);

  const atLimit = chefs.length >= MAX_CHEFS;

  async function handleResolve() {
    const trimmed = input.trim();
    if (!trimmed) return;

    setIsResolving(true);
    setResolveErr(null);
    setPreview(null);

    const result = await resolveChannel({ input: trimmed });

    setIsResolving(false);

    if (!result.ok) {
      setResolveErr(result.error);
      return;
    }

    setPreview({
      channelId:        result.channelId,
      channelName:      result.channelName,
      channelThumbnail: result.channelThumbnail,
      resolvedAt:       result.resolvedAt,
    });
  }

  async function handleAdd() {
    if (!preview) return;
    setAddError(null);

    try {
      await addChef({ sessionId, ...preview });
      setInput("");
      setPreview(null);
    } catch (err: unknown) {
      const msg = err instanceof Error ? err.message : "";
      if (msg === "duplicate") {
        setAddError("That chef is already in your list");
      } else if (msg === "limit_reached") {
        setAddError("You've reached the 6-chef limit");
      } else {
        setAddError("Something went wrong — try again");
      }
    }
  }

  async function handleRemove(channelId: string) {
    await removeChef({ sessionId, channelId });
  }

  return (
    <div className="min-h-screen bg-[#FAF6F1] pb-24">
      <div className="max-w-xl mx-auto px-4 py-8">

        <Link href="/" className="text-[#D4622A] text-sm mb-6 block hover:underline">
          ← Back to search
        </Link>

        <h1 className="text-2xl font-bold text-[#1A3A2A] mb-1">My Chefs</h1>
        <p className="text-gray-500 text-sm mb-6">
          Up to {MAX_CHEFS} YouTube cooking channels
        </p>

        {/* Saved chef list */}
        {chefs.length > 0 && (
          <div className="space-y-2 mb-6">
            {chefs.map((chef) => (
              <CustomChefCard
                key={chef.channelId}
                chef={chef}
                onRemove={() => handleRemove(chef.channelId)}
              />
            ))}
          </div>
        )}

        {/* Add section */}
        {atLimit ? (
          <p className="text-sm text-gray-500 text-center py-4">
            You&apos;ve reached the {MAX_CHEFS}-chef limit — remove one to add another.
          </p>
        ) : (
          <div className="space-y-3">
            <p className="text-sm font-semibold text-[#1A3A2A]">Add a chef</p>

            <div className="flex gap-2">
              <input
                type="text"
                value={input}
                onChange={(e) => {
                  setInput(e.target.value);
                  setResolveErr(null);
                  setPreview(null);
                }}
                onKeyDown={(e) => e.key === "Enter" && handleResolve()}
                placeholder="Paste YouTube channel URL or @handle"
                className="flex-1 border border-gray-200 rounded-2xl px-4 py-3 text-sm outline-none focus:border-[#D4622A] bg-white"
              />
              <button
                type="button"
                onClick={handleResolve}
                disabled={isResolving || !input.trim()}
                className="bg-[#D4622A] text-white px-5 py-3 rounded-2xl text-sm font-semibold disabled:opacity-50 hover:bg-[#c0521f] transition-colors"
              >
                {isResolving ? "..." : "Find"}
              </button>
            </div>

            {/* Resolve error */}
            {resolveErr && (
              <p className="text-red-600 text-sm">{errorMessage(resolveErr)}</p>
            )}

            {/* Preview card */}
            {preview && (
              <div>
                <CustomChefCard chef={preview} onAdd={handleAdd} />
                {addError && (
                  <p className="text-red-600 text-sm mt-1">{addError}</p>
                )}
              </div>
            )}
          </div>
        )}

        {/* Empty state */}
        {chefs.length === 0 && !atLimit && (
          <p className="text-xs text-gray-400 text-center pt-2">
            Add your first YouTube cooking channel above
          </p>
        )}
      </div>
    </div>
  );
}
```
